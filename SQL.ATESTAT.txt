
CREATE TABLE  "ASPECIALITATE" 
   (	"ID" NUMBER NOT NULL ENABLE, 
	"SPECIALITATE" VARCHAR2(70) NOT NULL ENABLE, 
	 CONSTRAINT "ASPECIALITATE_PK" PRIMARY KEY ("ID")
  USING INDEX  ENABLE
   )
/

CREATE SEQUENCE ASPECIALITATE_SEQ
 START WITH     1
 INCREMENT BY   1
 NOCACHE
 NOCYCLE;

CREATE OR REPLACE EDITIONABLE TRIGGER  "BI_ASPECIALITATE" 
  before insert on "ASPECIALITATE"               
  for each row  
begin   
  if :NEW."ID" is null then 
    select "ASPECIALITATE_SEQ".nextval into :NEW."ID" from sys.dual; 
  end if; 
end; 

/
ALTER TRIGGER  "BI_ASPECIALITATE" ENABLE
/
